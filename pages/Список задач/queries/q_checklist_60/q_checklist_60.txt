WITH p AS (
  SELECT
    NULLIF(CAST({{ selCompany.selectedOptionValue || '' }} AS text), '')::bigint AS company_id,
    NULLIF(CAST({{ selStatus.selectedOptionValue || '' }} AS text), '') AS status_text,
    NULLIF(CAST({{ selAssignee.selectedOptionValue || '' }} AS text), '')::uuid AS assignee_id
)
SELECT
  t.id AS "ID",
  c.name AS "Компания",
  t.title AS "Задача",

  CASE t.kind::text
    WHEN 'report' THEN 'Отчетность'
    WHEN 'payment' THEN 'Платеж'
    ELSE t.kind::text
  END AS "Тип",

  CASE t.status::text
    WHEN 'open' THEN 'Открыта'
    WHEN 'in_progress' THEN 'В работе'
    WHEN 'done' THEN 'Выполнена'
    WHEN 'overdue' THEN 'Просрочена'
    WHEN 'cancelled' THEN 'Отменена'
    ELSE t.status::text
  END AS "Статус",

  CASE t.priority::text
    WHEN 'low' THEN 'Низкий'
    WHEN 'medium' THEN 'Средний'
    WHEN 'high' THEN 'Высокий'
    WHEN 'critical' THEN 'Критичный'
    ELSE t.priority::text
  END AS "Приоритет",

  to_char(t.due_date::date, 'DD.MM.YYYY') AS "Срок",
  u.full_name AS "Ответственный",
  t.notes AS "Комментарий"

FROM tasks t
JOIN companies c ON c.id = t.company_id
JOIN users u ON u.id = t.assignee_user_id
CROSS JOIN p
WHERE c.is_active = TRUE
  AND t.due_date BETWEEN CURRENT_DATE AND CURRENT_DATE + INTERVAL '60 days'
  AND (p.company_id IS NULL OR c.id = p.company_id)
  AND (p.status_text IS NULL OR t.status::text = p.status_text)
  AND (p.assignee_id IS NULL OR u.id = p.assignee_id)
ORDER BY t.due_date ASC;
