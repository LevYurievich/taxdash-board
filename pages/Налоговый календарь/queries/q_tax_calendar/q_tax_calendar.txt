WITH p AS (
  SELECT
    COALESCE(
      NULLIF(CAST({{ dpFrom.selectedDate || '' }} AS text), '')::date,
      CURRENT_DATE
    ) AS date_from,
    COALESCE(
      NULLIF(CAST({{ dpTo.selectedDate || '' }} AS text), '')::date,
      (CURRENT_DATE + INTERVAL '60 days')::date
    ) AS date_to,
    NULLIF(CAST({{ selCalCompany.selectedOptionValue || '' }} AS text), '')::bigint AS company_id
)
SELECT
  t.id AS "ID",
  to_char(t.due_date::date, 'DD.MM.YYYY') AS "Срок",
  c.name AS "Компания",
  c.tax_regime AS "Режим",
  t.title AS "Событие",

  CASE t.kind::text
    WHEN 'report' THEN 'Отчетность'
    WHEN 'payment' THEN 'Платеж'
    ELSE t.kind::text
  END AS "Тип",

  CASE t.status::text
    WHEN 'open' THEN 'Открыта'
    WHEN 'in_progress' THEN 'В работе'
    WHEN 'done' THEN 'Выполнена'
    WHEN 'overdue' THEN 'Просрочена'
    WHEN 'cancelled' THEN 'Отменена'
    ELSE t.status::text
  END AS "Статус",

  u.full_name AS "Ответственный"
FROM tasks t
JOIN companies c ON c.id = t.company_id
JOIN users u ON u.id = t.assignee_user_id
CROSS JOIN p
WHERE c.is_active = TRUE
  AND t.due_date::date BETWEEN p.date_from AND p.date_to
  AND (p.company_id IS NULL OR c.id = p.company_id)
ORDER BY t.due_date::date ASC, c.name ASC, t.id ASC;
